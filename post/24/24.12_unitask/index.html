前言

在处理异步编程代码时，可以用到：C# 提供的 Task 类、Unity 使用 MonoBehaviour 提供的协程 Coroutine、Unity 社区提供的一个高性能 UniTask 库 ；

  • UniTask的说明：

为Unity提供一个高性能，0GC的async/await异步方案。

  1. 基于值类型的UniTask<T>和自定义的 AsyncMethodBuilder 来实现0GC
  2. 使所有 Unity 的 AsyncOperations 和 Coroutines 可等待
  3. 基于 PlayerLoop 的任务( UniTask.Yield, UniTask.Delay, UniTask.DelayFrame, etc…) 可以替换所有协程操作
  4. 对MonoBehaviour 消息事件和 uGUI 事件进行 可等待/异步枚举 拓展
  5. 完全在 Unity 的 PlayerLoop 上运行，因此不使用Thread，并且同样能在 WebGL、wasm 等平台上运行。
  6. 带有 Channel 和 AsyncReactiveProperty的异步 LINQ
  7. 提供一个 TaskTracker EditorWindow 以追踪所有UniTask分配来预防内存泄漏
  8. 与原生 Task/ValueTask/IValueTaskSource 高度兼容的行为

UniTask 功能依赖于 C# 7.0( task-like custom async method builder feature ) 所以需要的 Unity 最低版本是Unity 2018.3 ，官方支持的最低版本是Unity 2018.4.13f1.

  • 这三者的区别是：
特性 Unity协程 C# Task UniTask
限制 Unity MonoBehaviour .NET 通用 Unity 专用
返回类型 IEnumerator Task UniTask
性能 轻量 中等 高效
多线程支持 不支持 支持 支持
异步语法支持 不支持 支持（async/await） 支持（async/await）

    using System.Collections;
using Cysharp.Threading.Tasks;
using UnityEngine;

public class StudyUniTask : MonoBehaviour {
    async void Start() {
        // Unity 协程的写法
        StartCoroutine(Coroutine());
        // UniTask 也可对协程进行处理，等同以上写法
        //await Coroutine();
    }

    IEnumerator Coroutine() {
        Debug.Log("Start Coroutine ：" + Time.time);

        // Unity 协程等待 1 秒
        //yield return new WaitForSeconds(1);
        // 等待对应的毫秒时间, 1000 毫秒 = 1 秒
        yield return UniTask.Delay(1000).ToCoroutine();

        Debug.Log("Wait Coroutine ：" + Time.time);
    }
}

  3. 限制：引用文档

    您可以将协程（IEnumerator）转换为 UniTask（或直接等待），但它有一些限制。

    • 不支持WaitForEndOfFrame，WaitForFixedUpdate，Coroutine
    • Loop生命周期与StartCoroutine不一样，它使用指定PlayerLoopTiming的并且默认情况下，PlayerLoopTiming.Update在 MonoBehaviourUpdate和StartCoroutine的循环之前运行。

    如果您想要从协程到异步的完全兼容转换，请使用IEnumerator.ToUniTask(MonoBehaviour coroutineRunner)重载。它在参数 MonoBehaviour 的实例上执行 StartCoroutine 并等待它在 UniTask 中完成。

    using System.Threading;
using Cysharp.Threading.Tasks;
using UnityEngine;

public class StudyUniTask : MonoBehaviour {
    void Start() {
        // 因为此处不需要等待，因此调用 Forget 函数是为了消除 IDE 的警告提示
        WaitTest().Forget();
    }

    async UniTask WaitTest() {
        // 等待 Unity 渲染完每个摄像机和GUI后执行
        // 与 Unity WaitForEndOfFrame 用法相似，视版本需要添加参数 this
        await UniTask.WaitForEndOfFrame(this);
        Debug.Log("WaitForEndOfFrame");

        // 等待 FixedUpdate 函数结束后执行
        // 与 Unity WaitForFixedUpdate 用法一致
        await UniTask.WaitForFixedUpdate();
        Debug.Log("WaitForFixedUpdate");

        Debug.Log("当前帧数：" + Time.frameCount);
        // 等待到下一帧处理
        await UniTask.NextFrame();
        Debug.Log("NextFrame 帧数: " + Time.frameCount);

        // 等待到下一帧处理
        // 与 Unity yield return null 功能一致
        await UniTask.Yield();
        Debug.Log("Yield 帧数: " + Time.frameCount);

        // 等待对应的 Frame 数量
        await UniTask.DelayFrame(10);
        Debug.Log("DelayFrame 帧数：" + Time.frameCount);

        // 等待对应的毫秒时间, 1000 毫秒 = 1 秒
        // 知识点：参数内可用 TimeSpan 类转换成毫秒，如：TimeSpan.FromSeconds() 秒转换毫秒
        await UniTask.Delay(1000);
        Debug.Log("Delay");

        // 等待几秒后执行
        // 与 Unity WaitForSeconds 用法一致
        await UniTask.WaitForSeconds(1);
        Debug.Log("WaitForSeconds");

        // 指定某个值改变时触发
        await UniTask.WaitUntilValueChanged(this, x => x.isActive);
        Debug.Log("WaitUntilValueChanged");

        // 等待直到所提供的委托计算为 true
        // 与 Unity WaitUntil 用法一致
        await UniTask.WaitUntil(() => isActive == false);
        Debug.Log("WaitUntil");

        // 等待待直到所提供的委托计算为 false
        // 与 Unity WaitWhile 用法一致
        await UniTask.WaitWhile(() => isActive == false);
        Debug.Log("WaitWhile");

        // 等待直到某个 Token 被取消后
        await UniTask.WaitUntilCanceled(cts.Token);
        Debug.Log("WaitUntilCanceled");
    }

    public void Update() {
        if (Input.GetKeyDown(KeyCode.Q)) {
            // 按下后，按序触发 WaitUntilValueChanged 和 WaitUntil 和 WaitWhile
            isActive = !isActive;
        }else if (Input.GetKeyDown(KeyCode.W)) {
            // 按下后，执行触发 WaitUntilCanceled
            cts.Cancel();
        }
    }

    /// <summary>
    /// 是否激活
    /// </summary>
    private bool isActive = false;
    /// <summary>
    /// 取消任务源
    /// </summary>
    private CancellationTokenSource cts = new CancellationTokenSource();
}

    using Cysharp.Threading.Tasks;
using UnityEngine;

public class StudyUniTask : MonoBehaviour {
    void Start() {
        // 调用 Forget 函数是为了消除 IDE 的警告提示
        LoadResTest().Forget();
    }

    async UniTask LoadResTest() {
        Debug.Log("Time : " + Time.time);

        // 从 Resources 文件夹下加载名为 Res_1 的贴图
        await Resources.LoadAsync<Sprite>("Res_1");
        Debug.Log("Resources : " + Time.time);

        // 等待直到其中某个任务完成
        await UniTask.WhenAny(GetRes("Res_2"), GetRes("Res_3"));
        Debug.Log("WhenAny ：" + Time.time);

        // 等待直到任务全完成
        // 等待加载这2个资源后： Res_2 、 Res_3
        await UniTask.WhenAll(GetRes("Res_2"), GetRes("Res_3"));
        Debug.Log("WhenAll ：" + Time.time);

        Debug.Log("Void Start : " + Time.time);
        // 开启一个任务执行
        // 类似 .Net 的 += async () => { }
        UniTask.Void(async () => {
            Debug.Log("Void : " + Time.time);
            await Resources.LoadAsync<Sprite>("Res_1");
            Debug.Log("Void Await : " + Time.time);
        });

        Debug.Log("Defer Start : " + Time.time);
        // 开启一个任务执行，与上面用法一致，区别是不需要写 await
        await UniTask.Defer(async () => {
            Debug.Log("Defer : " + Time.time);
            await Resources.LoadAsync<Sprite>("Res_1");
            Debug.Log("Defer Await : " + Time.time);
        });
    }

    async UniTask<Sprite> GetRes(string iconName) {
        return await Resources.LoadAsync<Sprite>(iconName) as Sprite;
    }
}

using System;
using System.Threading;
using Cysharp.Threading.Tasks;
using UnityEngine;
using UnityEngine.UI;
using Object = UnityEngine.Object;

public class StudyUniTask : MonoBehaviour {

    public Image imgIcon;
    
    void Start() {
        // cts.Token 也可以写成 this.GetCancellationTokenOnDestroy()
        CancelTest(cts.Token).Forget();
        // 此代码会导致任务马上取消，因此会使图片加载失败
        cts.Cancel();
        // 设置多久以后导致任务取消
        // 目的是：当超时未完成任务后，主动取消任务，并做一些表现，可用于网络请求
        cts.CancelAfter(TimeSpan.FromSeconds(1));
    }

    async UniTask<string> CancelTest(CancellationToken token) {
        // WithCancellation 绑定了一个取消toke，SuppressCancellationThrow为了获取异常
        (bool isCanceled, Object result) = await Resources.LoadAsync<Sprite>("Res_1").WithCancellation(token).SuppressCancellationThrow();
        if (!isCanceled) {
            //imgIcon.sprite = result as Sprite;
            Debug.Log("图片加载成功");
        }
        else {
            Debug.Log("图片加载失败");
        }
        return "";
    }

    /// <summary>
    /// 取消任务源
    /// </summary>
    private CancellationTokenSource cts = new CancellationTokenSource();
}

using Cysharp.Threading.Tasks;
using UnityEngine;
using UnityEngine.UI;

public class StudyUniTask : MonoBehaviour {

    public Button btn;
    
    void Start() {
        Study().Forget();
    }

    async UniTask Study() {
        // 可以获取到 MonoBehaviour 的相关等待
        // this.GetAsync 
        // UGUI 上的也可以直接搜索 Async, 如下方 Button

        // 等待点击2次按钮

        await btn.OnClickAsync();
        Debug.Log("点击了2次");

        // 或者更高效的办法
        using var handler = btn.GetAsyncClickEventHandler();
        await handler.OnClickAsync();
        await handler.OnClickAsync();
        Debug.Log("点击了2次");
