设计思路

概括：

  1. 文本内容全在预制件、代码、配表中写。后续在翻译的时候用工具全导入插件内，并由工具导出给翻译人员，再导回到插件内。因此插件的作用是收集全部翻译文本，无需开发人员添加，统一文本位置，方便查找，方便导入导出；
  2. 资源内的多语言，则分为贴图资源和其他资源2个模块，这2个模块需要开发人员手动添加，因为只有用的额时候才能知道。且统一在这2个资源内，方便维护和查找；

文本

  1. 预制体上的文本

    • StringTablePrefab.asset：统一存放预制体文本的资源

    • 文本Key ：直接以当前文本作为 Key

      • 考虑的原因是：一开始想法是：预制体名 + 物体路径作为 key，但是考虑到不同预制体下会存在同文本的情况，这样会出现key不同，但Value一致的情况，导致重复翻译。因此改为文本作为key了；

    • LocalizaPrefab.cs：需要文本翻译的则挂载该脚本，只是为了存放Key；

    • LocalizationTools.cs：当前脚本中有个函数为，当打开预制件的时候，会往所需翻译的文本内添加该脚本；

      • 目的在于：为了支持切换多语言的时候，不需要重启游戏，且在切换不同语言的时候，文本会被本地化 Value 替换，因此需要有个 Key 存放的地方； ’

      • 支持：前缀 ` 标识的文本，不作为翻译内容；

      • 支持：字体的多语言文本翻译

        /// <summary>
/// 对预制体的文本进行翻译
/// </summary>
/// <param name="go"></param>
public static void LocalizationPrefabTxt(GameObject go, bool isModifyFont = false) {
    var txtArray = go.GetComponentsInChildren<Text>();
    foreach (var text in txtArray) {
        if (!IsNeedLocalizeTxt(text.text)) {
            continue;
        }
        var loPrefab = text.GetComponent<LocalizePrefab>();
        if (loPrefab == null) {
            loPrefab = text.gameObject.AddComponent<LocalizePrefab>();
            loPrefab.KeyId = text.text;
        }
        text.text = LocalizeMgr.GetText(loPrefab.KeyId, LocalizeMgr.EnumTable.StringPrefab);
        if (isModifyFont) {
            text.font = (Font) LocalizeMgr.GetRes("Font_Text");
        }
    }
    var txtMeshProArray = go.GetComponentsInChildren<TextMeshProUGUI>();
    foreach (var txtMeshPro in txtMeshProArray) {
        if (!IsNeedLocalizeTxt(txtMeshPro.text)) {
            continue;
        }
        var loPrefab = txtMeshPro.GetComponent<LocalizePrefab>();
        if (loPrefab == null) {
            loPrefab = txtMeshPro.gameObject.AddComponent<LocalizePrefab>();
            loPrefab.KeyId = txtMeshPro.text;
        }
        txtMeshPro.text = LocalizeMgr.GetText(loPrefab.KeyId, LocalizeMgr.EnumTable.StringPrefab);
        if (isModifyFont) {
            txtMeshPro.font = (TMP_FontAsset) LocalizeMgr.GetRes("Font_TextMeshPro");
        }
    }
}

  2. 代码中的文本

    • StringTableScript.asset：统一存放代码中需要翻译文本的资源

    • Key：开发人员自定义 string 类型的文本

      • 对应脚本：LocalizeConstScript.cs

        • 设计思路：
          1. 防止相同Key；
          2. 需要翻译的文本不存在各个脚本，统一在一个文件内；
          3. 统一个插件 key 的思路一致，都设置为 string 类型；
          4. 如果有机会，其他职能组改文本，也只需要到这个文件改；

  3. 表格中的文本

    • 方案1：
      • 如果本地中文表是Txt内容，并且代码可读了这个文本，则可以在不同语言内，生成名称一致但存放不同语言文件夹下的本地表，从而在切换不同的语言的时候，读取对应语言下的表内容；
      • 这个方案就与Localization插件隔离开了；
    • 方案2：
      • 在获取本地表后，举例：1张表的内容，以中文文本为Key，其他语言为Value，写一个工具，直接导入到Localization的对应配置表资源下；以此达到所有表的翻译
      • 这个方案就直接嵌入在Localization插件工具内

public class LocalizeMgr {
    /// <summary>
    /// 资源枚举
    /// </summary>
    public enum EnumTable {
        StringScript,
        StringPrefab,
    }
    /// <summary>
    /// 语言枚举表
    /// </summary>
    public enum EnumLocale {
        CN,
        EN,
    }

    /// <summary>
    /// 获取语言格式文本，具体文本在 ProjectSettings -> Localization -> Available Locales
    /// </summary>
    public static string GetLocaleFormatter(EnumLocale enumLocale) {
        switch (enumLocale) {
            case EnumLocale.EN:
                return "en";
            default:
                return "zh-CN";
        }
    }

    /// <summary>
    /// 获取多语言文本
    /// </summary>
    public static string GetText(string key, EnumTable enumTable = EnumTable.StringScript) {
        var locale = LocalizationSettings.Instance.GetSelectedLocale();
        var isCn = locale.Formatter.ToString() == GetLocaleFormatter(EnumLocale.CN);
        if (isCn && enumTable == EnumTable.StringScript) {
            return LocalizeConstScript.DicLocalization[key];
        }

        LocalizedStringTable stringTable = GetStringTableByEnum(enumTable);
        var entry = stringTable.GetTable().GetEntry(key);
        if (entry == null) {
            // 此处为了 CN环境下无需从表中读取
            if (!isCn) {
                Debug.LogWarning($"{stringTable.TableReference.TableCollectionName}:多语言包内没有Key：{key}");
            }
            return key;
        }
        string result = entry.GetLocalizedString();
        if (result == null) {
            Debug.LogWarning($"{stringTable.TableReference.TableCollectionName}:多语言包内Key：{key}的当前语言内没值");
            result = key;
        }
        return result;
    }

    /// <summary>
    /// 获取多语言资源
    /// </summary>
    public static Object GetRes(string key) {
        List<LocalizedAssetTable> assetList = new List<LocalizedAssetTable>{ assetTableCommon , assetTableSprite };
        AssetTableEntry entry = null;
        for (int i = 0; i < assetList.Count; i++) {
            entry = assetList[i].GetTable().GetEntry(key);
            if (entry != null) {
                break;
            }
        }
        if (entry == null) {
            Debug.LogWarning($"多语言包内没有Key：{key}");
            return null;
        }
        var path = AssetDatabase.GUIDToAssetPath(entry.Guid);
        return AssetDatabase.LoadAssetAtPath<Object>(path);
    }

    /// <summary>
    /// 获取字符串多语言包数据
    /// </summary>
    private static LocalizedStringTable GetStringTableByEnum(EnumTable enumTable) {
        LocalizedStringTable stringTable;
        switch (enumTable) {
            case EnumTable.StringPrefab:
                stringTable = stringTablePrefab;
                break;
            default:
                stringTable = stringTableScript;
                break;
        }
        return stringTable;
    }

    private static readonly LocalizedStringTable stringTableScript = new LocalizedStringTable { TableReference = LocalizeConst.StringTableScript };
    private static readonly LocalizedStringTable stringTablePrefab = new LocalizedStringTable { TableReference = LocalizeConst.StringTablePrefab };
    private static readonly LocalizedAssetTable assetTableCommon = new LocalizedAssetTable { TableReference = LocalizeConst.AssetTableCommon };
    private static readonly LocalizedAssetTable assetTableSprite = new LocalizedAssetTable { TableReference = LocalizeConst.AssetTableSprite };
}
