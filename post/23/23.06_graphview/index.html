基础类

  • 相关的API都在该命名空间下：UnityEditor.Experimental.GraphView
  • GraphView：渲染显示图的主类；
  • Node：一个节点；
  • Port：输入端口与输出端口；
  • Edge：端口与端口之间连接的线；
  • MiniMap：小地图，显示当前屏幕与整体节点位置信息；

public class GraphEditorTest : EditorWindow {
    [MenuItem("Tools/GraphView")]
    public static void Open() {
        GetWindow<GraphEditorTest>("GraphView");
    }

    void OnEnable() {
        graphView = new GraphViewTest {
            name = "GraphViewClass"
        };
        // 让graphView铺满整个Editor窗口
        graphView.StretchToParentSize();
        // 向当前窗口添加图形视图
        rootVisualElement.Add(graphView);
    }

    void OnDisable() {
        // 向当前窗口移除图形视图
        rootVisualElement.Remove(graphView);
    }

    private GraphView graphView;
}

public class GraphViewTest : GraphView {
    public GraphViewTest() {
        InitGraphViewInfo();
    }
    /// 初始化信息
    private void InitGraphViewInfo() {
        // 允许对Graph进行Zoom in/out（效果是：可放大放小视野）
        SetupZoom(ContentZoomer.DefaultMinScale, ContentZoomer.DefaultMaxScale);
        // 允许拖拽Content（效果是：可上下左右拖界面）
        this.AddManipulator(new ContentDragger());
        // 允许选择当前节点的内容（效果是：可选物体）
        this.AddManipulator(new SelectionDragger());
        // GraphView允许进行框选（效果是：多选物体）
        this.AddManipulator(new RectangleSelector());
    }
}

public class GraphViewTest : GraphView {
    public GraphViewTest() {
        InitGraphViewInfo();
        CreateNodeInfo();
    }
    /// 创建多节点
    private void CreateNodeInfo() {
        foreach (var value in nodeValueList) {
            CreateOneNode(value);
        }
    }
	// 创建节点函数
    private NodeTest CreateOneNode(int value) {
        NodeTest node = new NodeTest() {
            value = value,
            title = "Node" + value
        };
        int ten = value / 10 + 1;
        int unit = value % 10;
        float xRect = vec2NodeXY.x + (vec2NodeSize.x + vec2NodePadding.x) * (ten - 1);
        float yRect = vec2NodeXY.y + (vec2NodeSize.y + vec2NodePadding.y) * (unit - 1);
        node.SetPosition(new Rect(xRect, yRect, vec2NodeSize.x, vec2NodeSize.y));
        AddElement(node);
        return node;
    }
    /// 定义的节点假数据
    private readonly List<int> nodeValueList = new List<int>() {
        1, 11, 12, 13, 22
    };
    /// 节点的宽高
    private readonly Vector2 vec2NodeSize = new Vector2(140, 100);
    /// 节点初始 X、Y值
    private readonly Vector2 vec2NodeXY = new Vector2(100, 100);
    /// 节点之间的间隔
    private readonly Vector2 vec2NodePadding = new Vector2(100, 50);
}

    /// 判断哪些端口可以进行连接
    public override List<Port> GetCompatiblePorts(Port startPort, NodeAdapter adapter) {
        var compatiblePorts = new List<Port>();
        foreach (var port in ports.ToList()) {
            // 输入只能输出、端口自己不能连接自己，
            if (startPort.node == port.node ||
                startPort.direction == port.direction ||
                startPort.portType != port.portType) {
                continue;
            }

            compatiblePorts.Add(port);
        }
        return compatiblePorts;
    }

这时候是可以进行连接的，但是发现因为我用的是灰色背景，连接线也是灰色，导致会看不到线，因此需要更改下连接线的颜色，更改办法则是更改端口的颜色，代码会自动从输入端口过度颜色到输出端口

public class GraphEditorTest : EditorWindow {
    void OnEnable() {
        // 省略添加 GraphView 代码

        // 添加小地图
        minMap = new MiniMap();
        graphView.Add(minMap);
    }

    void OnDisable() {
        isInitMinMap = false;
        rootVisualElement.Remove(graphView);
    }

    void Update() {
        if (!isInitMinMap && graphView.contentRect.width > 0) {
            // 为了设置小地图在右上角，需要等待一帧才能获取到宽度
            int width = 300;
            minMap.SetPosition(new Rect(graphView.contentRect.width - width, 0, width, 200));
            isInitMinMap = true;
        }
    }

    private bool isInitMinMap = false;
    private MiniMap minMap;
    private GraphView graphView;
}
