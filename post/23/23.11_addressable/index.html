工具准备

使用的工具版本列表：

  • Unity：Unity 2021.3.6f1c1
  • IDE：Visual Studio Community 2019
  • Addressables：1.19.19
  • USO CDN

概述

本文实现的内容有：

  1. 游戏内使用 Addressables 作为AB资源管理；

  2. 游戏内接入 UOS CDN 作为远程服务资源热更；

  3. 初始包内只有启动预制件，玩家在游戏内初次游玩不同的关卡，需向远端下载资源才可进入关卡游玩；

  4. 远端下载的资源可下载、可使用、可删除；

  5. Gif 演示如下：

    • 演示顺序说明：
      • 演示1：打开游戏，此时关卡1与关卡2的资源处在远端并未下载；
      • 演示2：点击关卡1，进行远端资源下载，并在下载成功后，打开关卡1的预制件；
      • 演示3：回退主界面，重复点击关卡1的，因之前关卡1资源已经下载，因此直接打开关卡1；
      • 演示4：回退主界面，点击关卡1旁边的X按钮，删除关卡1的本地资源。再次点击关卡1，此时需要重新下载远端资源后才能打开关卡1；
      • 演示5：回退主界面，点击关卡2，因为关卡1与关卡2并不是同一远端资源，因此需要远端资源后才能打开关卡2；

    using UnityEngine;
using UnityEngine.AddressableAssets;
using UnityEngine.ResourceManagement.AsyncOperations;

public class Main : MonoBehaviour {
    // 热更地址
    // 选择关卡地址
    public string strAddressLevel;
    // 加载回调
    private AsyncOperationHandle<GameObject> handle;

    void Start() {
        StartLoad();
    }

    // 开始加载游戏界面
    public void StartLoad() {
        handle = Addressables.LoadAssetAsync<GameObject>(strAddressLevel);
        handle.Completed += HandleCompleted;
    }

    private void HandleCompleted(AsyncOperationHandle<GameObject> obj) {
        if (obj.Status != AsyncOperationStatus.Succeeded) {
            return;
        }
        Instantiate(obj.Result, transform);
        Addressables.Release(handle);
    }
}

    using System;
using System.Collections;
using UnityEngine;
using UnityEngine.AddressableAssets;
using UnityEngine.ResourceManagement.AsyncOperations;
using UnityEngine.UI;

public class SelectLevel : MonoBehaviour {
    public enum EnumLevel {
        Level1 = 1,
        Level2 = 2
    }
    public Button BtnLevel1;
    public Button BtnLevel2;
    public Button BtnDelete1;
    public Button BtnDelete2;
    public GameObject GoProgressInfo;
    public Text TxtSize;
    public Text TxtProgress;
    public Slider SldProgress;
    private AsyncOperationHandle<GameObject> handleInstance;

    void Start() {
        AddListeners();
        DownLoadProgressInfo(false);
        StartCoroutine(RenderBtnInfo(EnumLevel.Level1));
        StartCoroutine(RenderBtnInfo(EnumLevel.Level2));
    }
    private void AddListeners() {
        BtnLevel1.onClick.AddListener(() => { OnClickBtnLevel(EnumLevel.Level1); });
        BtnLevel2.onClick.AddListener(() => { OnClickBtnLevel(EnumLevel.Level2); });
        BtnDelete1.onClick.AddListener(() => { OnClickBtnDelete(EnumLevel.Level1); });
        BtnDelete2.onClick.AddListener(() => { OnClickBtnDelete(EnumLevel.Level2); });
    }

    #region Addressable 函数
    /// <summary>
    /// 释放实例化的资源
    /// </summary>
    public void ReleaseHandleInStance() {
        Addressables.Release(handleInstance);
    }
    /// <summary>
    /// 点击了关卡按钮
    /// </summary>
    private void OnClickBtnLevel(EnumLevel enumLevel) {
        StartCoroutine(DownloadLevelCache(enumLevel));
    }
    /// <summary>
    /// 清除下载缓存
    /// </summary>
    private void OnClickBtnDelete(EnumLevel enumLevel) {
        var key = GetKeyByEnumLevel(enumLevel);
        Addressables.ClearDependencyCacheAsync(key);
        StartCoroutine(RenderBtnInfo(enumLevel));
    }
    /// <summary>
    /// 下载关卡缓存
    /// </summary>
    private IEnumerator DownloadLevelCache(EnumLevel enumLevel) {
        var key = GetKeyByEnumLevel(enumLevel);
        var handleDown = Addressables.GetDownloadSizeAsync(key);
        yield return handleDown;
        SetDownLoadSize(handleDown.Result);
        if (handleDown.Result > 0) {
            DownLoadProgressInfo(true);
            yield return new WaitForSeconds(1);
        }
        var handle = Addressables.DownloadDependenciesAsync(key, false);
        while (!handle.IsDone) {
           var status = handle.GetDownloadStatus();
           float progress = status.Percent;
           if (progress > 0) {
               DownLoadProgressInfo(true, progress);
           }
           yield return null;
        }
        yield return handle.IsDone;
        Addressables.Release(handle);
        DownLoadProgressInfo(false);
        handleInstance = Addressables.InstantiateAsync(key, transform.parent);
        handleInstance.Completed += (obj) => {
            StartCoroutine(RenderBtnInfo(enumLevel));
        };
    }

    private string GetKeyByEnumLevel(EnumLevel enumLevel) {
        string str = "";
        switch (enumLevel) {
            case EnumLevel.Level1:
                str = "Assets/Prefabs/Level/Level1.prefab";
                break;
            case EnumLevel.Level2:
                str = "Assets/Prefabs/Level/Level2.prefab";
                break;
        }
        return str;
    }
    #endregion

    #region UI 函数
    /// <summary>
    /// 渲染按钮的下载状态
    /// </summary>
    private IEnumerator RenderBtnInfo(EnumLevel enumLevel) {
        var key = GetKeyByEnumLevel(enumLevel);
        var handleDown = Addressables.GetDownloadSizeAsync(key);
        yield return handleDown;
        bool hasDown = handleDown.Result <= 0;
        switch (enumLevel) {
            case EnumLevel.Level1:
                BtnDelete1.gameObject.SetActive(hasDown);
                break;
            case EnumLevel.Level2:
                BtnDelete2.gameObject.SetActive(hasDown);
                break;
        }
        Addressables.Release(handleDown);
    }
    /// <summary>
    /// 渲染下载信息
    /// </summary>
    private void DownLoadProgressInfo(bool isShow, float progress = 0f) {
        GoProgressInfo.SetActive(isShow);
        if (!isShow) {
            return;
        }
        TxtProgress.text = progress * 100 + "%";
        SldProgress.value = progress;
    }
    /// <summary>
    /// 设置下载大小
    /// </summary>
    private void SetDownLoadSize(long size) {
        TxtSize.text = "远程下载：" + ByteConversion(size);
    }
    private string ByteConversion(double length) {
        var k = 1024;
        var m = k * 1024;
        var g = m * 1024;
        string temp;
        if (length < k) {
            temp = Math.Round(length, 0) + "B";
        } else if (length < m) {
            temp = Math.Round(length / (k), 0) + "KB";
        } else if (length < g) {
            temp = Math.Round(length / (m), 2) + "MB";
        } else {
            temp = Math.Round(length / (g), 2) + "GB";
        }
        return temp;
    }
    #endregion
}

Addressable

  1. Addressables 设置：
    • （可参考上面文章链接，以下为简化步骤）
    • 初始化创建：
      • 清空创建：Addressables Groups -> Build -> Clean Build -> All
      • 新增创角：Addressables Groups -> Build -> New Build -> Default Build Script
    • 划分远程与本地组：
      • 将 Level 相关的预制件放在远程组，将 启动预制（SelectLevel） 相关的预制件放在本地组。这样做的目的是：只更新远程组，本地组为最小启动包；
    • 真实模拟远程包设置：
      • Addressables Groups -> PlayModeScript -> UseExistingBuild
  2. USO CDN 设置：
    • 按照上面文章链接操作一遍便可

调试

  1. 打开引擎运行游戏，即可看到效果；
  2. 原理为：
    • 点击关卡按钮的时候；
    • Addressables.GetDownloadSizeAsync(）获取需要下载的大小；
    • Addressables.DownloadDependenciesAsync() 判断是否下载成功，是则进入关卡，否则进入远端下载流程；
    • 下载成功后，会把远端的下载资源存在本地；
