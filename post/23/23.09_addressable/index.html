    using UnityEngine;
using UnityEngine.AddressableAssets;
using UnityEngine.ResourceManagement.AsyncOperations;

public class Test : MonoBehaviour {
    // 加载资源地址
    public string strAddress;
    // 加载回调
    private AsyncOperationHandle<GameObject> handle;

    void Start() {
        if (strAddress.Length <= 0) {
            Debug.LogError("StrAddress is Null");
            return;
        }
        handle = Addressables.LoadAssetAsync<GameObject>(strAddress);
        handle.Completed += HandleCompleted;
    }
    void OnDestroy() {
        Addressables.Release(handle);
    }
    private void HandleCompleted(AsyncOperationHandle<GameObject> obj) {
        if (obj.Status != AsyncOperationStatus.Succeeded) {
            Debug.LogError($"Asset for {strAddress} failed to load.");
            return;
        }
        Instantiate(obj.Result, transform);
    }
}

Addressable 设置

  • 创建 Addressable

    • 菜单路径：Windows -> Asset Management -> Addressables -> Groups
    • Groups 窗口：点击 Create Addressables Settings 生成 Addressables 配置

  • 标记预制体为 Addressable

    • 点击 GoInstance 预制件，在 Inspector 上方点击 Addressable 标记，并把标记成功后的地址填写到 Canvas 上挂载 Test 脚本的引用上

  • 创建远程服务：

    • 菜单路径：Windows -> Asset Management -> Addressables -> Hosting
    • Hosting 窗口：Create -> Local Hosting，生成成功后，点击右方的 Enable 开启服务；

  • 设置远程配置：

    • AddressableAssetSettings 设置：

      • Project路径：AddressableAssetSettings -> Build Remote Catalog -> Remote 开启远程

    • AddressableGroups 设置：

      • 菜单路径：Windows -> Asset Management -> Addressables -> Groups

      • Groups 窗口：

        • 选择 Play Mode Script -> Use Existing Build：从上次的内容构建中加载内容
        • 选择 Build -> New Build -> Default Build Script : 生成一个默认的更新脚本

      • Inspector 操作：在 Groups 窗口内对准 Default Local Group（这个名称如果有设置的话，则点击设置的对应组） 双击，系统会自动选择 Project 上的相对应配置，并打开 Inspector后

        • 选择 Update Restriction 为 Cannot Change Post Release
        • 选择 Build & Load Paths 为 Remote

  • 运行游戏：此时能正常实例化预制体出来，则是正确的；

打包

  • 菜单路径：File -> Build Settings -> 对准 Windows 平台点击 Build，生成 PC 包

  • 验证：运行 PC 包，查看PC包内的展示的内容是否与 Unity 内展示是否一致；

热更预制体

  • 更改预制体：点击 GoInstance 预制件，更改 Image 颜色 与 Text 文本

  • 运行游戏：发现还是旧的预制件，那是因为 Groups 窗口内我们选择的是：Play Mode Script -> Use Existing Build

  • 更新预制体：

    • 打开 Groups 窗口，点击 Tools -> Check for Content Update Restrictions -> 选择 Windows 文件夹 -> addressables_content_state.bin 文件
    • 此时会打开 Content Update Preview 窗口，当前窗口会列出 热更的资源，当前只有 GoInstance 预制件，然后点击 Apply Changes

  • 更新生成文件：

    • 打开 Addressable Groups 窗口，点击 Build -> Update a Previous Build -> 选择 Windows 文件夹 -> addressables_content_state.bin 文件

    • 此时 Groups 窗口下的 GoInstance 的所处位置从 Default Local Group 转到了 Content Update 路径下，Content Update 路径下表示的是需要热更的资源

  • 热更验证：

    • 此时运行游戏 或者 运行刚才已经打好的PC包（不需要重新打包）

二、热更界面

通常手游的热更流程是：启动游戏 -> 判断是否有热更资源 -> 没有则进入游戏 -> 有则启动热更界面 -> 告知玩家热更大小并开始从远程服务端下载资源 -> 下载成功后则进入游戏
