前言

本文主要介绍屏蔽词过滤功能，主要应用在角色昵称命名和聊天内容上做提醒和过滤；

功能分析

  • 需要有个屏蔽字库，里面存放着需要屏蔽的词汇；
  • 代码需要基础实现：
    1. 判断是否有屏蔽词汇；
    2. 替换屏蔽词汇为 * ；
    3. 需要有个高效一点的查找屏蔽词汇方式；
      • 利用 Trie 树做查找和插入比较高效，该树使用空间换时间，使用字符串的公共前缀达到提高查询效率的目的；
      • 详细介绍，请看 参考链接；

using System.Text;
using System.Collections.Generic;
 
namespace Core {
    public class ChatTrieNode{
        public bool IsWord { get; set; }
        public Dictionary<char, ChatTrieNode> Children { get; set; }
        public ChatTrieNode() {
            Children = new Dictionary<char, ChatTrieNode>();
        }
    }

    public class ChatFilter {
        private ChatTrieNode root;
        public ChatFilter() {
            root = new ChatTrieNode();
        }
 
        /// <summary>
        /// 添加屏蔽词
        /// </summary>
        public void AddWord(string word) {
            ChatTrieNode current = root;
            foreach (char c in word) {
                if (!current.Children.ContainsKey(c)) {
                    current.Children.Add(c, new ChatTrieNode());
                }
                current = current.Children[c];
            }
            current.IsWord = true;
        }
 
        /// <summary>
        /// 判断词是否存在
        /// </summary>
        public bool ContainsWord(string word) {
            ChatTrieNode current = root;
            foreach (char c in word) {
                if (!current.Children.ContainsKey(c)) {
                    return false;
                }
                current = current.Children[c];
            }
            return current.IsWord;
        }
 
        /// <summary>
        /// 判断给定消息是否包含屏蔽词
        /// </summary>
        public bool ContainsFilterWords(string message) {
            int index = 0;
            while (index < message.Length) {
                int start = index;
                ChatTrieNode current = root;
                while (index < message.Length && current.Children.ContainsKey(message[index])) {
                    current = current.Children[message[index++]];
                    // 只要匹配到一个就返回
                    if (current.IsWord) {
                        return true;
                    }
                }
                index = ++start;
            }
            return false;
        }
 
        /// <summary>
        ///  将给定的消息其中的屏蔽词替换为 *
        /// </summary>
        public string ReplaceFilterWords(string message) {
            int index = 0;
            StringBuilder sb = new StringBuilder();
            while (index < message.Length) {
                int start = index;
                int end = -1;
                ChatTrieNode current = root;
                while (index < message.Length && current.Children.ContainsKey(message[index])) {
                    current = current.Children[message[index++]];
                    // 找到最长的连续屏蔽词
                    if (current.IsWord) {
                        end = index - 1;
                    }
                }
                // 有找到屏蔽词
                if (end != -1) {
                    while (start <= end) {
                        start++;
                        sb.Append("*");
                    }
                    index = start;
                } else {
                    sb.Append(message[start++]);
                    index = start;
                }
            }
            return sb.ToString();
        }
    }
}
